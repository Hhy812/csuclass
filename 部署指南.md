## 一、部署到自有服务器

### 方式1：静态文件部署（推荐）

#### 1.1 获取构建文件

构建完成后，所有需要的文件都在 `dist/` 目录中：

```
dist/
├── index.html          # 入口文件
├── assets/             # 静态资源
│   ├── index-xxx.js    # JS文件
│   └── index-xxx.css   # CSS文件
└── classroom_data.json # 教室数据
```

#### 1.2 部署到常见服务器

##### Nginx 部署

```nginx
server {
    listen 80;
    server_name your-domain.com;
    root /var/www/classroom-query/dist;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # 启用gzip压缩
    gzip on;
    gzip_types text/plain text/css application/json application/javascript;
}
```

##### Apache 部署

创建 `.htaccess` 文件：

```apache
<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /
  RewriteRule ^index\.html$ - [L]
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule . /index.html [L]
</IfModule>
```

##### 腾讯云/阿里云 OSS + CDN

1. 创建 OSS Bucket
2. 上传 `dist/` 目录下所有文件
3. 配置静态网站托管
4. 绑定自定义域名（可选）

```bash
# 使用 ossutil 上传
ossutil cp -r dist/ oss://your-bucket-name/
```

##### GitHub Pages

```bash
# 1. 创建 GitHub 仓库
# 2. 推送 dist 文件夹到 gh-pages 分支
git checkout --orphan gh-pages
git rm -rf .
cp -r dist/* .
git add .
git commit -m "Deploy to GitHub Pages"
git push origin gh-pages
```

#### 1.3 Docker 部署

创建 `Dockerfile`：

```dockerfile
FROM nginx:alpine
COPY dist/ /usr/share/nginx/html/
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
```

创建 `nginx.conf`：

```nginx
server {
    listen 80;
    location / {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri $uri/ /index.html;
    }
}
```

构建并运行：

```bash
docker build -t classroom-query .
docker run -d -p 8080:80 classroom-query
```

## 二、更新教室数据

### 2.1 数据文件格式

数据文件 `classroom_data.json` 格式：

```json
{
  "classrooms": {
    "A101": {
      "building": "A",
      "floor": 1,
      "schedule": {
        "1": {
          "周一": {
            "1-2节": "free",
            "3-4节": "free"
          }
        }
      }
    }
  },
  "buildings": {
    "A": {
      "1": ["A101", "A102", "A103"],
      "2": ["A201", "A202"]
    }
  }
}
```

### 2.2 从Excel更新数据

使用以下Python脚本转换新的Excel文件：

```python
import pandas as pd
import json
from collections import defaultdict

def convert_excel_to_json(excel_path, output_path):
    """
    将空闲教室Excel转换为JSON格式
    
    Excel格式要求：
    - 每个周一个Sheet，命名为"第1周"、"第2周"等
    - 每行是一个时间段（1-2节、3-4节等）
    - 每列是一个星期（周一、周二等）
    - 单元格内容是空闲教室列表，空格分隔
    """
    
    xl = pd.ExcelFile(excel_path)
    
    classroom_data = {
        'classrooms': defaultdict(lambda: {
            'building': '',
            'floor': 0,
            'schedule': defaultdict(lambda: defaultdict(dict))
        }),
        'buildings': defaultdict(lambda: defaultdict(list))
    }
    
    days = ['周一', '周二', '周三', '周四', '周五']
    time_slots = ['1-2节', '3-4节', '5-6节', '7-8节', '9-10节']
    
    for sheet_name in xl.sheet_names:
        week_num = int(sheet_name.replace('第', '').replace('周', ''))
        df = pd.read_excel(excel_path, sheet_name=sheet_name, header=None)
        
        for col_idx in range(1, 6):  # 周一到周五
            day = days[col_idx - 1]
            for row_idx in range(2, 7):  # 5个时间段
                time_slot = time_slots[row_idx - 2]
                
                cell_value = df.iloc[row_idx, col_idx]
                if pd.notna(cell_value):
                    classrooms = str(cell_value).strip().split()
                    
                    for classroom in classrooms:
                        classroom = classroom.strip()
                        if len(classroom) >= 3:
                            # 解析教学楼和楼层
                            building = classroom[0] if classroom[0] in 'ABCD' else 'other'
                            try:
                                floor = int(classroom[1])
                            except:
                                floor = 1
                            
                            # 更新数据
                            classroom_data['classrooms'][classroom]['building'] = building
                            classroom_data['classrooms'][classroom]['floor'] = floor
                            classroom_data['classrooms'][classroom]['schedule'][str(week_num)][day][time_slot] = 'free'
                            
                            # 更新楼层索引
                            if classroom not in classroom_data['buildings'][building][str(floor)]:
                                classroom_data['buildings'][building][str(floor)].append(classroom)
    
    # 转换为普通dict
    result = {
        'classrooms': dict(classroom_data['classrooms']),
        'buildings': {k: dict(v) for k, v in classroom_data['buildings'].items()}
    }
    
    # 排序
    for building in result['buildings']:
        for floor in result['buildings'][building]:
            result['buildings'][building][floor] = sorted(result['buildings'][building][floor])
    
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(result, f, ensure_ascii=False, indent=2)
    
    print(f"数据已保存到 {output_path}")
    print(f"共 {len(result['classrooms'])} 个教室")
    
    return result

# 使用示例
if __name__ == '__main__':
    convert_excel_to_json('新学期空闲教室.xlsx', 'classroom_data.json')
```

### 2.3 更新部署

1. 运行转换脚本生成新的 `classroom_data.json`
2. 将新文件上传到服务器的 `dist/` 目录
3. 刷新页面即可看到新数据

```bash
# 上传到服务器
scp classroom_data.json user@your-server:/var/www/classroom-query/dist/

# 或使用 rsync
rsync -avz classroom_data.json user@your-server:/var/www/classroom-query/dist/
```

### 3.2 使用SVG绘制精确平面图

创建 `src/components/FloorPlanSVG.tsx`：

```tsx
import React from 'react';

interface FloorPlanSVGProps {
  building: string;
  floor: number;
  classrooms: string[];
  statuses: Record<string, 'free' | 'occupied'>;
  onClassroomClick: (classroom: string) => void;
}

export function FloorPlanSVG({ building, floor, classrooms, statuses, onClassroomClick }: FloorPlanSVGProps) {
  // 根据实际平面图定义教室位置和大小
  const classroomPositions: Record<string, Record<number, Array<{
    id: string;
    x: number;
    y: number;
    width: number;
    height: number;
  }>>> = {
    'A': {
      1: [
        // 根据实际平面图定义
        { id: 'A101', x: 50, y: 50, width: 80, height: 60 },
        { id: 'A102', x: 150, y: 50, width: 80, height: 60 },
        // ... 更多教室
      ]
    }
  };

  const positions = classroomPositions[building]?.[floor] || [];

  return (
    <svg viewBox="0 0 800 600" className="w-full h-auto">
      {/* 绘制走廊 */}
      <rect x="0" y="250" width="800" height="100" fill="#e2e8f0" />
      <text x="400" y="305" textAnchor="middle" fill="#94a3b8" fontSize="16">走廊</text>
      
      {/* 绘制教室 */}
      {positions.map((pos) => {
        const status = statuses[pos.id] || 'occupied';
        const color = status === 'free' ? '#10b981' : '#f43f5e';
        
        return (
          <g key={pos.id} onClick={() => onClassroomClick(pos.id)} style={{ cursor: 'pointer' }}>
            <rect
              x={pos.x}
              y={pos.y}
              width={pos.width}
              height={pos.height}
              fill={color}
              stroke="white"
              strokeWidth="2"
              rx="4"
            />
            <text
              x={pos.x + pos.width / 2}
              y={pos.y + pos.height / 2}
              textAnchor="middle"
              dominantBaseline="middle"
              fill="white"
              fontSize="14"
              fontWeight="bold"
            >
              {pos.id}
            </text>
          </g>
        );
      })}
      
      {/* 绘制楼梯、电梯等 */}
      <rect x="350" y="20" width="40" height="40" fill="#cbd5e1" />
      <text x="370" y="45" textAnchor="middle" fontSize="12">楼梯</text>
    </svg>
  );
}
```

### 3.3 获取精确平面图的方法
